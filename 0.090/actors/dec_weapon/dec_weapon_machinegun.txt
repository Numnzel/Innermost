ACTOR Machinegun : Weapon {

	//$NotAngled
	//$Category Decorate/Weapons
	//$Title "Light machinegun"
	//$Sprite MCGNW0

	Weapon.Kickback 180
	Weapon.AmmoType "NewClip"
	Weapon.AmmoGive 50
	Weapon.AmmoUse 1
	Weapon.Kickback 200
	Weapon.SlotNumber 4
	Weapon.SlotPriority 0.0
	Weapon.SelectionOrder 1

	+WEAPON.NOAUTOAIM

	Inventory.PickupSound "misc/w_pkup"
	Inventory.PickupMessage "Picked up a machinegun!"
	Obituary "%o was lit up by %k's Machinegun."
 	Decal BulletChip

	States {

		Ready:
			MCGN A 1 A_WeaponReady
			Loop
		Deselect:
			MCGN A 1 A_Lower
			Loop
		Select:
			MCGN A 1 A_Raise
			Loop
		Fire:
			TNT1 A 0 A_PlaySound("weapons/Machinegun", CHAN_WEAPON, 0.7)
			MCGN B 3 A_FireBullets(1.4, 1.4, -1, 7, "BulletPuff")
			MCGN C 3
			MCGN A 3
			TNT1 A 0 A_Refire
			goto Ready
		Spawn:
			MCGN W -1
			Stop
	}
}
