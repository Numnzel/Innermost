ACTOR HeavyRifle : Weapon 15060 {
	
	//$NotAngled
	//$Category Decorate/Weapons
	//$Title "Heavy rifle"
	//$Sprite HVYRA0
	
	Inventory.PickupMessage "You got a heavy rifle!"
	Inventory.PickupSound "misc/w_pkup"
	
	Weapon.AmmoType "HeavyClip"
	Weapon.AmmoGive 10
	Weapon.AmmoUse 1
	Weapon.Kickback 230
	Weapon.SlotNumber 5
	Weapon.SlotPriority 1.0
	
	Decal BulletChip
	
	+WEAPON.NOAUTOFIRE
	+WEAPON.NOAUTOAIM
	
	States {
		
		Spawn:
			HVYR A -1
			Stop
		Ready:
			HVRG A 1 A_WeaponReady
			Loop
		Deselect:
			HVRG A 1 A_Lower
			Loop
		Select:
			HVRG A 0 A_PlaySound("weapons/rifleup")
			HVRG A 1 A_Raise
			Goto Select+1
		Fire:
			HVRF B 0 Bright A_PlaySound("weapons/riflefre", CHAN_WEAPON)
			HVRF B 3 Bright A_FireBullets(0.8, 0.8, -1, random(49, 50), "BulletPuff") // random(32, 45), "BulletPuff")
			HVRG AAA 3
			HVRG A 0 A_PlaySound("weapons/shotgr")
			HVRG AAAAA 3
			Goto Ready
	}
}
