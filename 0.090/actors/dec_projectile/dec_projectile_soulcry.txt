Actor SoulCry {
	
	Decal Scorch
	Scale 0.8
	Radius 3
	Height 6
	Speed 20
	
	// MaxStepHeight 24
	
	PROJECTILE
	-NOGRAVITY
	+NOEXPLODEFLOOR
	// +FLOORHUGGER
	+THRUACTORS
	
	SeeSound "weapons/rocklf"
	DeathSound "weapons/rocklx"
	
	Obituary "%o couldn't withstand the soul cry damage."
	
	States {
		
		Spawn:
			SCRY A 8 BRIGHT
		Explosion:
			SCRY A 1 A_JumpIf (z - floorz == 0, 1)
			Loop
			TNT1 A 0 A_SpawnItemEx ("SoulCryExplosion")//, 0, 0, 0, 0, 0, 0, SXF_NOCHECKPOSITION)
			Goto Spawn
		Death:
			TNT1 A 0 A_SpawnItemEx ("SoulCryExplosion")//, 0, 0, 0, 0, 0, 0, SXF_NOCHECKPOSITION)
			Stop
	}
}
